// 台股存摺 Cloudflare Worker v10 // 基於 v9，移除所得稅，新增去年殖利率 const CORS = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, OPTIONS', 'Content-Type': 'application/json; charset=utf-8', }; const memCache = new Map(); // 支援上市(.TW)和上櫃(.TWO) function yahooSyms(code) { if (/[A-Z]/.test(code)) return [`${code}.TWO`, `${code}.TW`]; return [`${code}.TW`, `${code}.TWO`]; } export default { async fetch(req) { if (req.method === 'OPTIONS') return new Response(null, { headers: CORS }); const url = new URL(req.url); const code = (url.searchParams.get('code') || '').toUpperCase().trim(); const path = url.pathname; try { if (path === '/api/health') return ok({ status:'ok', time:new Date().toISOString() }); if (path === '/api/price') return await apiPrice(code); if (path === '/api/exright') return await apiExRight(code); if (path === '/api/stock') return await apiStock(code); if (path === '/api/debug') return await apiDebug(code); return err('找不到路由', 404); } catch(e) { return err('錯誤：' + e.message, 500); } } }; // ══════════════════════════════════════ // 取得配息歷史（自動試 .TW / .TWO） // ══════════════════════════════════════ async function fetchDividends(code) { const now = Math.floor(Date.now()/1000); const past = now - 365*3*24*3600; // 3 年 for (const sym of yahooSyms(code)) { try { const r = await yFetch(`https://query1.finance.yahoo.com/v8/finance/chart/${sym}?interval=1d&period1=${past}&period2=${now}&events=dividends`); const json = await r.json(); const divEvents = json?.chart?.result?.[0]?.events?.dividends || {}; const records = Object.values(divEvents) .map(ev => ({ exDate: tsToDate(ev.date), payDate: '', dividend: +(ev.amount || 0), year: new Date(ev.date*1000).getFullYear(), })) .sort((a,b) => b.exDate.localeCompare(a.exDate)); if (records.length > 0) return { records, sym }; } catch(e) {} } return { records: [], sym: '' }; } // ══════════════════════════════════════ // 計算頻率、今年/去年殖利率統計 // ══════════════════════════════════════ function calcStats(all, price) { const cur = new Date().getFullYear(); const prev = cur - 1; // 去年固定 const byYear = {}; all.forEach(r => { if (!byYear[r.year]) byYear[r.year] = []; byYear[r.year].push(r); }); const thisData = byYear[cur] || []; const prevData = byYear[prev] || []; // 用近12個月資料推算頻率（適合新ETF） const cutoff = new Date(); cutoff.setMonth(cutoff.getMonth() - 12); const recent = all.filter(r => new Date(r.exDate) >= cutoff); const freq = recent.length >= 10 ? '月配' : recent.length >= 6 ? '月配' : recent.length >= 3 ? '季配' : recent.length >= 1 ? '年配' : '不定期'; // 今年累計配息 const thisSrc = thisData.length ? thisData : prevData; const annDiv = +thisSrc.reduce((s,r) => s+(r.dividend||0), 0).toFixed(4); // 去年累計配息（固定用 cur-1） const prevDiv = +prevData.reduce((s,r) => s+(r.dividend||0), 0).toFixed(4); const prevYield = (price && prevDiv) ? +(prevDiv/price*100).toFixed(2) : 0; // 年化殖利率 const latest = all[0] || {}; const mult = freq==='月配'?12 : freq==='季配'?4 : 1; const curYield = (price && latest.dividend) ? +(latest.dividend/price*100*mult).toFixed(2) : 0; const annYield = (price && annDiv) ? +(annDiv/price*100).toFixed(2) : 0; return { freq, annDiv, prevDiv, thisYearCount: thisData.length, annualYield: annYield, currentYield: curYield, prevYearYield: prevYield, prevYear: prev, // 固定去年 }; } // ══════════════════════════════════════ // DEBUG // ══════════════════════════════════════ async function apiDebug(code) { if (!code) return err('請提供代號'); const { records, sym } = await fetchDividends(code); const priceData = await fetchPrice(code); const price = priceData?.price || 0; const stats = calcStats(records, price); return ok({ code, sym, records, price, stats }); } // ══════════════════════════════════════ // 1. 股價 // ══════════════════════════════════════ async function apiPrice(code) { if (!code) return err('請提供代號'); const hit = getCache('price:'+code); if (hit) return ok(hit); const d = await fetchPrice(code); if (!d) return err('查無股價'); setCache('price:'+code, d, 300); return ok(d); } async function fetchPrice(code) { for (const sym of yahooSyms(code)) { try { const r = await yFetch(`https://query1.finance.yahoo.com/v8/finance/chart/${sym}?interval=1d&range=1d`); const json = await r.json(); const meta = json?.chart?.result?.[0]?.meta; if (!meta?.regularMarketPrice) continue; const price = meta.regularMarketPrice; const prev = meta.chartPreviousClose || meta.previousClose || price; const change = prev ? +((price-prev)/prev*100).toFixed(2) : 0; return { code, sym, name: meta.longName || meta.shortName || code, price, change, high: meta.regularMarketDayHigh || price, low: meta.regularMarketDayLow || price, volume: meta.regularMarketVolume || 0, source: 'Yahoo Finance', }; } catch(e) {} } return null; } // ══════════════════════════════════════ // 2. 除權息 // ══════════════════════════════════════ async function apiExRight(code) { if (!code) return err('請提供代號'); const hit = getCache('ex:'+code); if (hit) return ok(hit); const { records, sym } = await fetchDividends(code); const priceData = await fetchPrice(code); const price = priceData?.price || 0; const stats = calcStats(records, price); const latest = records[0] || {}; const d = { code, sym, freq: stats.freq, records: records.slice(0, 24), annualDividend: stats.annDiv, prevYearDividend: stats.prevDiv, latestDividend: latest.dividend || 0, latestExDate: latest.exDate || '', latestPayDate: '', nextExDate: estimateNext(latest.exDate, stats.freq), thisYearCount: stats.thisYearCount, annualYield: stats.annualYield, currentYield: stats.currentYield, prevYearYield: stats.prevYearYield, prevYear: stats.prevYear, // 是否需要手動輸入（Yahoo 查不到記錄） needsManualInput: records.length === 0, source: records.length ? `Yahoo Finance (${sym})，共 ${records.length} 筆` : `查無配息記錄（請手動輸入）`, }; if (records.length > 0) setCache('ex:'+code, d, 3600); return ok(d); } // ══════════════════════════════════════ // 3. 整合查詢 // ══════════════════════════════════════ async function apiStock(code) { if (!code) return err('請提供代號'); const [priceData, exRes] = await Promise.all([ fetchPrice(code), apiExRight(code), ]); const ex = (await exRes.json()).data; if (!priceData) return err('查無此代號：' + code); const price = priceData.price || 0; const div = ex?.latestDividend || 0; return ok({ code, name: priceData.name || code, type: isETF(code) ? 'ETF' : '股票', freq: ex?.freq || '不定期', price, change: priceData.change || 0, high: priceData.high || 0, low: priceData.low || 0, volume: priceData.volume || 0, // 配息資料 dividend: div, annualDividend: ex?.annualDividend || 0, prevYearDividend: ex?.prevYearDividend || 0, // 殖利率（三種） currentYield: ex?.currentYield || 0, // 年化（最新配息×頻率÷股價） annualYield: ex?.annualYield || 0, // 今年累計÷股價 prevYearYield: ex?.prevYearYield || 0, // 去年累計÷股價 prevYear: ex?.prevYear || 0, exDate: ex?.latestExDate || '', payDate: ex?.latestPayDate || '', nextExDate: ex?.nextExDate || '', dividendHistory: ex?.records || [], thisYearCount: ex?.thisYearCount || 0, // 是否需要手動輸入 needsManualInput: ex?.needsManualInput || false, incomeRatio: 100, updatedAt: new Date().toISOString(), source: ex?.source || 'Yahoo Finance', }); } // ══════════════════════════════════════ // 工具 // ══════════════════════════════════════ function yFetch(url) { return fetch(url, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 Chrome/120.0', 'Accept': 'application/json', 'Referer': 'https://finance.yahoo.com/', } }); } function tsToDate(ts) { const d = new Date((ts + 8*3600) * 1000); return d.toISOString().slice(0, 10); } function isETF(c) { return /^(0[0-9]{3,5}|006[0-9]{3})$/.test(c); } function estimateNext(last, freq) { if (!last) return ''; const days = { 月配:30, 季配:91, 年配:365, 不定期:365 }; const d = new Date(last); if (isNaN(d.getTime())) return ''; const step = days[freq] || 365; d.setDate(d.getDate() + step); const today = new Date(); while (d < today) d.setDate(d.getDate() + step); return d.toISOString().slice(0, 10); } function getCache(k) { const i = memCache.get(k); if (!i || Date.now() > i.exp) { memCache.delete(k); return null; } return i.data; } function setCache(k,v,t) { memCache.set(k, { data:v, exp:Date.now()+t*1000 }); } function ok(d) { return new Response(JSON.stringify({ok:true, data:d, ts:Date.now()}), {headers:CORS}); } function err(m,s=400) { return new Response(JSON.stringify({ok:false, error:m, ts:Date.now()}), {status:s, headers:CORS}); }
